Description¶

In this notebook we will pick the cleansed csv file. This cleansed csv file was generated from the notebook in which we did the EDA & data preprocessing (compaign-EDA.ipynb).Since the target labels are highly imbalanced, we will use Over Sampling techniques, in this case SMOTE. Post that we will do the following:
We will first remove the features which have shown less important and focus on the top 20 features only

1. Run a baselined model using Logistic Regression
2. Run a Decision Tree model & also fine tune params using GridSearchCV
3. Run a RandomForest model & also fine tune params using GridSearchCV
4. Run a AdaBoost model
5. Run a XGBoost model
6. Run a Neural Network based model 

For each model we will print the performance metrics like 
1. Accuracy
2. Precision
3. Recall
4. F1-score
5. Classification Report
6. ROC curve


Finally we print a summary of performance for different models

def prepare_x_and_y(df, y):
    
    df['Target'] = y
    df.loc[df['Target'] == 'no', 'Target'] = 0
    df.loc[df['Target'] == 'yes', 'Target'] = 1

    #data_orig_wt = data_orig_wt.dropna(how='any',axis=0)

    y = df['Target']
    df = df.drop('Target',axis=1)

    print ('Size of dataset after encoding target is ', df.shape)

    trainX, testX, trainY, testY = train_test_split(df, y)
    smt = SMOTE()
    trainX, trainY = smt.fit_sample(trainX, trainY)
    
    return df, trainX, testX, trainY, testY

from sklearn.metrics import accuracy_score, f1_score, precision_score, recall_score
from sklearn.metrics import confusion_matrix, classification_report

from sklearn.metrics import roc_curve

model_names = []
model_accuracy = []
model_f1_score = []
model_recall_score = []
model_precision_score = []
model_roc_threshold_score = []

def show_roc_curve(falsePositiveRate, truePositiveRate):
    plt.plot(falsePositiveRate, truePositiveRate)
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.plot([0,1],[0,1],'r--')
    plt.xlim([-0.001,1])
    plt.ylim([0,1.001])
    plt.show()
    
def show_model_performance (model_name, predY, testY):
    
    print ('Accuracy Score is ', accuracy_score(testY, predY))
    print ('F1 Score is ', f1_score(testY, predY))
    print ('Precision Score is ', precision_score(testY, predY))
    print ('Recall Score is ', recall_score(testY, predY))
    
    print (confusion_matrix(testY, predY))
    print (classification_report(testY, predY))   
    
    fpr, tpr, threshold = roc_curve(testY, predY)
    sns.set(rc={'figure.figsize':(10,5)})
    show_roc_curve(fpr, tpr)
    
    model_names.append(model_name)
    model_accuracy.append(accuracy_score(testY, predY))
    model_f1_score.append(f1_score(testY, predY))
    model_recall_score.append(recall_score(testY, predY))
    model_precision_score.append(precision_score(testY, predY))
    model_roc_threshold_score.append(threshold)
    
def show_feature_importance(model, df):

    #Print the feature importance of the decision model
    feature_imp = model.feature_importances_
    features_imp_df = pd.DataFrame(feature_imp, columns = ["Imp"], index = df.columns)
    print (features_imp_df.sort_values(by=['Imp'],ascending=False))

    sns.set(rc={'figure.figsize':(31.7,18.27)})
    sns.barplot(df.columns, feature_imp, orient='v')

import xgboost as xb

def use_xgboost(modelName, df, p_trainX, p_testX, p_trainY, p_testY):
    testX_matrix= p_testX.values
    testY_matrix= p_testY.values

    dTrainMatrix = xb.DMatrix(p_trainX, p_trainY)
    dTestMatrix = xb.DMatrix(testX_matrix, testY_matrix)

    xb_params = {"learning_rate":"0.01", "objective":"binary:logistic","n_estimators":"1000","alpha":"1"}
    model = xb.train(params=xb_params, dtrain=dTrainMatrix, num_boost_round=5)
    predictedY = model.predict(data=dTestMatrix)

    predY = []
    for p in predictedY:
        if p>0.5:
            predY.append(1)
        else :
            predY.append(0)

            #show model performance metrics
    show_model_performance(modelName, predY, p_testY)
